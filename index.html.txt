    // 1) Theme toggle (light/dark) with localStorage
    (function(){
      const root = document.documentElement;
      const btn = document.getElementById('themeToggle');
      const saved = localStorage.getItem('mm_theme');
      if(saved === 'dark') {
        document.documentElement.setAttribute('data-theme','dark');
        btn.textContent = '‚òÄÔ∏è Light';
      } else {
        document.documentElement.removeAttribute('data-theme');
        btn.textContent = 'üåó Dark';
      }
      btn.addEventListener('click', ()=>{
        const isDark = document.documentElement.getAttribute('data-theme') === 'dark';
        if(isDark){
          document.documentElement.removeAttribute('data-theme');
          localStorage.setItem('mm_theme','light');
          btn.textContent = 'üåó Dark';
        } else {
          document.documentElement.setAttribute('data-theme','dark');
          localStorage.setItem('mm_theme','dark');
          btn.textContent = '‚òÄÔ∏è Light';
        }
      });
    })();

    // 2) Simple slideshow logic (supports multiple images later)
    (function(){
      const images = ['photo.jpg']; // agar aage images add karoge toh yahan 'slide2.jpg','slide3.jpg' add kar dena
      const imgEl = document.getElementById('heroImage');
      let idx = 0;
      if(images.length > 1){
        setInterval(()=>{
          idx = (idx + 1) % images.length;
          imgEl.style.opacity = 0;
          setTimeout(()=>{ imgEl.src = images[idx]; imgEl.style.opacity = 1; }, 400);
        }, 5000);
      } else {
        // subtle slow zoom on single image
        imgEl.style.transition = 'transform 12s ease';
        setInterval(()=> {
          imgEl.style.transform = imgEl.style.transform === 'scale(1.06)' ? 'scale(1)' : 'scale(1.06)';
        }, 6500);
      }
    })();
